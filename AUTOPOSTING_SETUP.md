# ----------------------
# Telegram Bot
# ----------------------
# Как получить:
# 1. Найдите @BotFather в Telegram
# 2. Отправьте команду /newbot
# 3. Следуйте инструкциям и получите токен
# 4. Добавьте бота в свой канал как администратора
# 5. ID канала начинается с @ (например: @yourchannel) или -100xxxxxxxxxx для приватных каналов

TELEGRAM_BOT_TOKEN=your_bot_token_here
TELEGRAM_CHANNEL_ID=@yourchannel


# ----------------------
# VK API
# ----------------------
# Как получить:
# 1. Перейдите на https://vk.com/apps?act=manage
# 2. Создайте новое Standalone приложение
# 3. Перейдите в Настройки → Токены
# 4. Создайте ключ доступа с правами:
#    - photos (загрузка фото)
#    - wall (публикация на стене)
#    - groups (управление сообществом)
# 5. ID группы - это число из адреса группы (vk.com/club12345 → 12345)

VK_ACCESS_TOKEN=your_vk_access_token_here
VK_GROUP_ID=12345
